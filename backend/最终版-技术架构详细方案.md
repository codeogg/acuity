### 2.5 React Aria 国际化日期组件

```tsx
// components/shared/DatePicker.tsx
import { DatePicker, DateField, DateSegment, Label, Group } from "react-aria-components";
import { parseDate } from "@internationalized/date";

interface Props {
  value?: string;              // ISO格式 "2026-07-03"
  onChange: (isoDate: string) => void;
  label: string;
}

export function AppDatePicker({ value, onChange, label }: Props) {
  return (
    <DatePicker
      value={value ? parseDate(value) : null}
      onChange={(date) => date && onChange(date.toString())}
      className="flex flex-col gap-1"
    >
      <Label className="text-sm font-medium">{label}</Label>
      <Group className="flex rounded-md border border-input px-2 py-1.5">
        <DateField>
          {(segment) => <DateSegment segment={segment} className="px-0.5 tabular-nums" />}
        </DateField>
      </Group>
    </DatePicker>
  );
}
```

**选型理由**：香港场景需支持分段式键盘输入(年/月/日独立输入)，React Aria在a11y与IME兼容性上比`react-day-picker`更成熟，适合医生快速录入场景。

### 2.6 PDF标注画布（核心难点组件）

```tsx
// components/annotator/PdfCanvas.tsx
"use client";
import * as pdfjsLib from "pdfjs-dist";

interface FieldBoxData {
  id: string;
  pageNo: number;
  x: number; y: number; width: number; height: number;  // PDF坐标系(pt,左下原点)
  fieldType: "text" | "checkbox" | "date" | "signature" | "image";
  mappedStandardField?: string;
  rowVersion: number;    // 对应后端乐观锁字段
}

// 坐标转换：PDF坐标系(左下原点,pt) <-> 屏幕坐标系(左上原点,px)
function pdfToScreen(pdfY: number, fieldHeight: number, pageHeight: number, scale: number) {
  return (pageHeight - pdfY - fieldHeight) * scale;
}
```

**交互要点**：
- pdf.js渲染PDF页面到`<canvas>`，字段框用绝对定位`<div>`叠加层
- 拖拽/缩放建议用`react-rnd`或自研pointer events处理
- 保存字段位置调整时携带`rowVersion`，后端乐观锁校验冲突（返回409时前端提示"已被他人修改，请刷新"）

### 2.7 数据获取策略

- **Server Component优先**：列表/详情页直接在Server Component中调用FastAPI
- **React Query**：标注工具、AI核对表单等强交互场景用于mutation+乐观更新
- **前后端类型同步**：CI中自动执行 `openapi-typescript http://api/openapi.json -o lib/api/generated/schema.d.ts`，配合`openapi-fetch`生成强类型客户端，杜绝手写interface导致的字段不一致

### 3.3 Vertex AI Gemini 接入（已确认方案）

**鉴权方式**：走 Google Cloud 服务账号(Service Account) + IAM权限，而非消费级API Key，满足企业生产环境的审计与权限管控要求。

```python
# src/modules/ai_extraction/gemini_client.py
from google import genai
from google.genai import types
from src.config import settings

class GeminiClient:
    def __init__(self):
        self.client = genai.Client(
            vertexai=True,
            project=settings.GCP_PROJECT_ID,
            location=settings.GCP_LOCATION,   # asia-east1(台湾)/asia-southeast1(新加坡)，就近香港
        )

    async def extract_structured(self, prompt: str, response_schema: dict) -> dict:
        """病历文本结构化提取：用response_schema强约束JSON输出，避免解析失败"""
        response = await self.client.aio.models.generate_content(
            model="gemini-2.5-flash",
            contents=prompt,
            config=types.GenerateContentConfig(
                response_mime_type="application/json",
                response_schema=response_schema,
                temperature=0.1,               # 医疗信息提取场景需低随机性
            ),
        )
        return {
            "parsed": response.parsed,
            "token_usage": response.usage_metadata.total_token_count,
        }

    async def analyze_pdf_page_image(self, image_bytes: bytes, hint_prompt: str) -> dict:
        """PDF页面视觉字段识别：用于模板标注AI辅助"""
        response = await self.client.aio.models.generate_content(
            model="gemini-2.5-pro",             # 视觉理解任务用pro提升准确率
            contents=[
                types.Part.from_bytes(data=image_bytes, mime_type="image/png"),
                hint_prompt,
            ],
            config=types.GenerateContentConfig(
                response_mime_type="application/json",
                temperature=0.1,
            ),
        )
        return response.parsed
```

**环境变量：**
```bash
# .env
GCP_PROJECT_ID=your-project-id
GCP_LOCATION=asia-east1
GOOGLE_APPLICATION_CREDENTIALS=/secrets/service-account.json
```

**模型选型：**
| 场景 | 模型 | 原因 |
|---|---|---|
| 病历文本提取 | gemini-2.5-flash | 纯文本任务，速度快成本低 |
| PDF扫描件视觉识别 | gemini-2.5-pro | 视觉+版面理解精度更高 |

**弹性与降级：**
- 超时10s，失败重试2次(指数退避)
- Redis限流：每诊所每分钟最多N次调用
- AI不可用时不阻塞医生流程，前端提示"AI识别暂不可用，请手动填写"

**生产环境待办清单（上线前必须完成）：**
1. 开通GCP项目 + 启用 Vertex AI API
2. 创建服务账号，授予 `roles/aiplatform.user` 权限
3. 与Google签署DPA(数据处理协议)，明确香港病历数据处理边界（涉及PDPO合规，需法务对接）
4. 确认 Gemini 2.5系列模型在目标region(如asia-east1)的可用性（新模型有时先在us-central1上线，需查阅Vertex AI region支持列表）
5. 密钥管理：生产环境不将service-account.json打入镜像，采用K8s Secret挂载或GCP Workload Identity

### 3.4 动态Prompt构建

```python
# src/modules/ai_extraction/prompt_builder.py
def build_extraction_prompt(text: str, required_fields: list[StandardField]) -> tuple[str, dict]:
    field_descriptions = "\n".join(
        f"- {f.field_code}（{f.field_name}）：{f.ai_extraction_hint or '从病历中提取对应信息'}"
        for f in required_fields
    )
    prompt = f"""你是医疗保险单据助理。请从以下病历文本中提取信息。
需要提取的字段：
{field_descriptions}

病历原文：
\"\"\"
{text}
\"\"\"

若某字段在病历中找不到依据，value填null，禁止编造信息。
"""
    response_schema = {
        "type": "object",
        "properties": {
            f.field_code: {
                "type": "object",
                "properties": {
                    "value": {"type": ["string", "null"]},
                    "confidence": {"type": "number"},
                },
                "required": ["value", "confidence"],
            }
            for f in required_fields
        },
    }
    return prompt, response_schema
```

```python
class TemplateField(Base):
```

### 3.8 认证鉴权

```python
# src/core/security.py
from jose import jwt
from datetime import datetime, timedelta, UTC
from src.config import settings

def create_access_token(user_id: int, role: str, clinic_id: int | None = None) -> str:
    payload = {
        "sub": str(user_id),
        "role": role,
        "clinic_id": clinic_id,
        "exp": datetime.now(UTC) + timedelta(hours=8),
    }
    return jwt.encode(payload, settings.JWT_SECRET, algorithm="HS256")
```

- Token存于httpOnly Secure Cookie，`middleware.ts`做路由级校验，FastAPI依赖注入层做接口级二次校验
- 所有DOCTOR角色接口强制注入`clinic_id`过滤（见功能规格模块5数据隔离规则）

### 3.9 病历文本字段级加密

```python
# src/core/encryption.py
from cryptography.fernet import Fernet
from src.config import settings

fernet = Fernet(settings.FIELD_ENCRYPTION_KEY)  # 密钥建议走KMS/Secrets Manager管理

def encrypt_text(plain: str) -> str:
    return fernet.encrypt(plain.encode()).decode()

def decrypt_text(cipher: str) -> str:
    return fernet.decrypt(cipher.encode()).decode()
```

`claim_submission.medical_record_text` 存储前加密，读取时解密，数据库层不可见明文，满足PDPO合规要求。
